rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Public website (calculator/contact forms) can create leads,
    // but only the signed-in admin can read, update, or delete them.
    match /leads/{leadId} {
      allow create: if true;
      allow read, update, delete: if request.auth != null;
    }

    // Proposals, surveys and tasks are only ever touched from the CRM
    // dashboard, so they always require an authenticated admin.
    match /proposals/{proposalId} {
      allow read, write: if request.auth != null;
    }
    match /surveys/{surveyId} {
      allow read, write: if request.auth != null;
    }
    match /tasks/{taskId} {
      allow read, write: if request.auth != null;
    }
  }
}
